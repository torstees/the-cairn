{# Shared alias-name renderers for tune listings. Both macros share the same hover tooltip box (via _alias_tooltip_names, private to this file) so the truncation/tooltip markup only exists in one place. alias_summary(aliases): "Also known as: ..." — used where a tune's row already has room for a second line (tunes/, boxes/{id}, lists/{id}, progress/). alias_badge(aliases): a compact "aka" trigger with the same hover tooltip but no visible name list — used where the row layout has no room for a second line (sets/{id}'s single-line, numbered member rows). Both render inline and take a list of objects with a `.name` attribute (i.e. TuneAlias instances, or anything shaped like one) — callers wrap the call in whatever block-level spacing fits their own layout. Usage: {% from "components/_alias_summary.html" import alias_summary, alias_badge %} {{ alias_summary(tune.aliases) }} {{ alias_badge(tune.aliases) }} #} {% macro _alias_tooltip_names(names, align="left") %} {{ names | join(', ') }} {% endmacro %} {% macro alias_summary(aliases, max_shown=5) %} {% if aliases %} {% set names = aliases | map(attribute='name') | list %} Also known as: {{ names[:max_shown] | join(', ') }}{% if names | length > max_shown %}…{% endif %} {% if names | length > max_shown %}{{ _alias_tooltip_names(names) }}{% endif %} {% endif %} {% endmacro %} {% macro alias_badge(aliases) %} {% if aliases %} {% set names = aliases | map(attribute='name') | list %} {# Right-anchored, unlike alias_summary's tooltip: this trigger always sits at the end (shrink-0) of a row, so a left-anchored tooltip would routinely overflow off the right edge of the viewport. #} aka {{ _alias_tooltip_names(names, align="right") }} {% endif %} {% endmacro %}
