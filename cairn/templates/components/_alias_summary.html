{# Shared "Also known as: ..." renderer for tune listings (tunes/, boxes/{id}, lists/{id}, progress/). Shows the first `max_shown` alias names inline, followed by "…" and a hover tooltip listing all of them when there are more than that. Callers import this macro and wrap the call in whatever block-level spacing fits their own layout (e.g. a
) — this macro itself renders inline so it drops into an existing line cleanly. Usage: {% from "components/_alias_summary.html" import alias_summary %} {{ alias_summary(tune.aliases) }} #} {% macro alias_summary(aliases, max_shown=5) %} {% if aliases %} {% set names = aliases | map(attribute='name') | list %} Also known as: {{ names[:max_shown] | join(', ') }}{% if names | length > max_shown %}…{% endif %} {% if names | length > max_shown %} {{ names | join(', ') }} {% endif %} {% endif %} {% endmacro %}
